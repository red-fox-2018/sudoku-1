function cek kolom (parameter : board, row, currValue)
  LOOP col sebanyak 9
    IF board row col == currValue
      return false
    END IF
  END LOOP

function cek baris (parameter : board, col, currValue)
  LOOP row sebanyak 9
    IF board row col == currValue
      return false
    END IF
  END LOOP
  return true

///////////////////////
function cek blok 3x3 (parameter: board, row, col, currValue)
  dapatkan angka row dengan (row/3)*3
  dapatkan angka col dengan (col/3)*3

  LOOP i sebanyak 3
    LOOP j sebanyak 3
      IF board row+i col+j == currValue
        return false
      END IF
    END LOOP
  END LOOP
